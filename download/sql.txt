SELECT dev_type AS device_type, dev_id AS device_id, IF(!dev_notrack AND !dev_noident,ac_type,"" ) AS aircraft_model, IF(!dev_notrack AND !dev_noident,air_acreg,"") AS registration, IF(!dev_notrack AND !dev_noident,air_accn,"") AS cn, IF(!dev_notrack,"Y","N") AS tracked, IF(!dev_noident,"Y","N") AS identified FROM devices, aircraftstypes, trackedobjects where air_actype = ac_id and dev_flyobj = air_id ORDER BY dev_id ASC;
